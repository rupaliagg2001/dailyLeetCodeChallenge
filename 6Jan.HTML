1833 Maximum Ice Cream Bars:

Link- Maximum Ice Cream Bars

class Solution {
    public:
        int maxIceCream(vector& costs, int coins) {
            sort(costs.begin(), costs.end());
            int count = 0;
            for(int i = 0; i < costs.size(); i++){
                if(costs[i] <= coins){
                    count++;
                    coins -= costs[i];
                }
            }
        return count;
    }
};
