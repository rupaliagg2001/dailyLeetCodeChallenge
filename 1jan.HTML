Link- Word Pattern

class Solution {
    public:
    bool wordPattern(string pattern, string s) {
        unordered_mapmp;
        vectorvec;
        string temp = "";
        for(int i = 0; i < s.size(); ++i){
            if(s[i] == ' '){
                vec.push_back(temp);
                temp = "";
            }
            else temp += s[i];
        }
        vec.push_back(temp);
        setst1;
        setst2;
        for(int i = 0; i < vec.size(); ++i){
            st1.insert(pattern[i]);
            st2.insert(vec[i]);
        }
        if(st1.size() != st2.size() || vec.size() != pattern.size()){
            return false;
        }
        for(int i = 0; i < vec.size(); ++i){
            if(mp.find(pattern[i]) != mp.end()){
                if(mp[pattern[i]] != vec[i]){
                    return false;
                }else continue;
            }
            else{
                mp[pattern[i]] = vec[i];
            }
        
        }
        return true;
    }
};
