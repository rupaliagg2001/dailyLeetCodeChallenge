Link- Binary Tree Preorder Traversal

class Solution {
    public:
        void preorder(TreeNode* root, vector&vec){
            if(root == NULL){
                return;
            }
            vec.push_back(root->val);
            preorder(root->left, vec);
            preorder(root->right, vec);
        }
        vector preorderTraversal(TreeNode* root) {
            vectorvec;
            preorder(root, vec);
            return vec;
        }
    };
