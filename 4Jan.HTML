2244 Minimum Rounds to Complete All Tasks:

Link- Minimum Rounds to Complete All Tasks

class Solution {
public:
    int minimumRounds(vector& tasks) {
        mapmp;
        for(int i = 0; i < tasks.size(); i++){
            mp[tasks[i]]++;
        }
        int count = 0;
        for(auto it : mp){
            if(it.second < 2){
                return -1;
            }
            else if(it.second == 2 || it.second == 3){
                count++;
            }
            else if(it.second > 3){
                if(it.second % 3 == 0){
                    count += it.second / 3;
                }
                else{
                    count += it.second / 3 + 1;
                }
            }
        }
        return count;
    }
};
