944 Delete Columns to Make Sorted:

Link- Delete Columns to Make Sorted

class Solution {
    public:
        int minDeletionSize(vector& strs) {
            int n = strs[0].length();
            int count = 0;
            for(int i = 0; i < n; i++){
                for(int j = 0; j < strs.size() - 1; j++){
                    if(strs[j][i] > strs[j + 1][i]){
                        count++;
                        break;
                    }
                }
        }
        return count;
    }
};
