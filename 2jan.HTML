class Solution {
    public:
    bool detectCapitalUse(string word) {
        int n = word.length(); 
        if(isupper(word[0])){
            if(isupper(word[1])){
            for(int i = 2; i < n; i++){
                if(islower(word[i])){
                    return false;
                }
            }
            }
            else if(islower(word[1])){
                for(int i = 2; i < n; i++){
                if(isupper(word[i])){
                    return false;
                }
            }  
            }
        }
        else if(islower(word[0])){
            for(int i = 1; i < n; i++){
                if(isupper(word[i])){
                    return false;
                }
        }
        }
        return true;
    }
};
